### 백준 토마토(3차원)


#### 3차원 코너케이스 
> 2차원으로 풀 경우 층이 바뀌는 경우에 예외처리를 해주어야 한다. 
> 예> 세로가 3인 경우 제일 위와 제일 아래는 다음 층의 1층, 이전층의 3층과 분리되어야 한다.

```python
def emptyAll() :
    for queue in first_ripe :
        if queue :
            return False
    return True

def all_Ripen() :
    for i in range(N*H) :
        for j in range(M) :
            if not box[i][j] :
                return False
    return True

from collections import deque

M, N, H = map(int, input().split())
box = [list(map(int, input().split())) for _ in range(N*H)]
result = 0
if all_Ripen() : print(result)
else :
    first_ripe = deque([deque([(i,j)]) for i in range(N*H) for j in range(M) if box[i][j]==1])
    while not emptyAll() and not all_Ripen():
        
        for queue in first_ripe :
            for _ in range(len(queue)) :
                q = queue.popleft()
                x,y = q[0],q[1]
                
                if x % N == 0 :
                    for dx, dy in (1, 0), (0, 1), (0, -1), (N, 0), (-N, 0):
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < N * H and 0 <= ny < M and not box[nx][ny]:
                            queue.append([nx, ny])
                            box[nx][ny] = 1
                elif x % N == N-1 :
                    for dx, dy in (0, 1), (-1, 0), (0, -1), (N, 0), (-N, 0):
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < N * H and 0 <= ny < M and not box[nx][ny]:
                            queue.append([nx, ny])
                            box[nx][ny] = 1
                else : 
                    for dx, dy in (1,0),(0,1),(-1,0),(0,-1),(N,0),(-N,0) :
                        nx, ny = x+dx, y+dy
                        if 0 <= nx < N*H and 0 <= ny < M and not box[nx][ny]:
                            queue.append([nx,ny])
                            box[nx][ny] = 1
        result += 1

    if all_Ripen() :
        print(result)
    else :
        print(-1)
```
