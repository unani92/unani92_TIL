```python
import sys

def findstart() :
    for i in range(N) :
        for j in range(N) :
            if not visited[i][j] :
                return i,j

def inFalse(visited) :
    for i in visited :
        if False in i :
            return True
    return False

N = int(sys.stdin.readline())
lst = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]
rain = list(set([a for i in lst for a in i]))
result = 1
for i in rain :
    visited = [[False] * N for _ in range(N)]

    for a in range(N) :
        for b in range(N) :
            if lst[a][b] <= i :
                visited[a][b] = True

    cnt = 0
    while inFalse(visited) :

        start = findstart()
        stack = [(start[0],start[1])]
        while stack :
            n = stack.pop()
            x,y = n[0],n[1]
            visited[x][y] = True
            for dx,dy in (1,0),(0,1),(-1,0),(0,-1) :
                test_x,test_y = x+dx,y+dy
                if 0<=test_x<N and 0<=test_y<N and not visited[test_x][test_y] :
                    stack.append((test_x,test_y))

        cnt += 1

    if cnt > result :
        result = cnt

print(result)
```
